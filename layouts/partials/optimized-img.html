{{- /* Renders a resized image from assets/images with optional WebP. Accepts: src (path under assets/images), alt, title, class, style, lazy, priority */ -}} {{- $src := .src | strings.TrimPrefix "/" | strings.TrimPrefix "assets/" | strings.TrimPrefix "images/" -}} {{- $alt := .alt | default "" -}} {{- $title := .title | default "" -}} {{- $class := .class | default "" -}} {{- $style := .style | default "" -}} {{- $lazy := .lazy | default true -}} {{- $priority := .priority | default false -}} {{- $maxWidth := .maxWidth | default 1200 -}} {{- $img := resources.Get (printf "images/%s" $src) -}} {{- if $img -}} {{- $resized := $img.Resize (printf "%dx" (int $maxWidth)) -}} {{- $webp := $resized.Process "webp" -}} {{- else -}} {{- $fallback := (printf "assets/%s" $src) | strings.TrimPrefix "/" | absURL -}} {{- end -}}
